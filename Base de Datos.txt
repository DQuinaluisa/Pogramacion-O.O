
-- Database: computadoras

-- DROP DATABASE computadoras;

CREATE DATABASE computadoras
    WITH 
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Spanish_Spain.1252'
    LC_CTYPE = 'Spanish_Spain.1252'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1;



--crear tabla
CREATE TABLE procesador(
	id Serial,
	modelo varchar (50),
	marca  varchar (10),
	nucleos varchar (10),
	socket  varchar (10),
	primary key (id)
)
